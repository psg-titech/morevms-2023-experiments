function [] = showhist(n)
   fd = mopen(n, 'rb');
   d = mget(120000, 'i', fd)/50;
   printf("max:%f", max(d));
   histplot(50, d);
   mclose(fd);
endfunction

men = mopen('emfrp-node1.dat', 'rb');
en = mget(120000, 'i', men);
mclose(men);
mean(en/50) // => 3.3862688
sum(3.35<=(en/50) & (en/50)<=3.43)/length(en) // => 0.995875
